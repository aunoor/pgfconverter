    #pragma pack(push,1)
    struct CFG_HEADER
    {
       uint8 nMajorVersion;
       uint8 nMinorVersion;
       uint16 nFieldCount;
       uint8 reserved[2];
    };
    #pragma pack(pop)

    struct CFG_USER_SAFETY_V1
    {
       int32         pos_x;
       int32         pos_y;
       unsigned char   type;
       unsigned char   speed;
       char         name[128];
    };

    enum CFG_USER_SAFETY_INFO
    {
       CFG_USER_SAFETY_INFO_TYPE_NONE               = 1,
       CFG_USER_SAFETY_INFO_TYPE_SPEEDLIMIT,
       CFG_USER_SAFETY_INFO_TYPE_SCHOOLZONE,
       CFG_USER_SAFETY_INFO_TYPE_DANGER,
       CFG_USER_SAFETY_INFO_TYPE_SPEEDBUMP,
       CFG_USER_SAFETY_INFO_TYPE_WILDLIFE,
       CFG_USER_SAFETY_INFO_TYPE_RAILWAY,
       CFG_USER_SAFETY_INFO_TYPE_POLICE,
       CFG_USER_SAFETY_INFO_TYPE_SPEEDCAM,
       CFG_USER_SAFETY_INFO_TYPE_DANGEROUS_INTERSECTION,
       CFG_USER_SAFETY_INFO_TYPE_DANGEROUS_TURN
    };


В начале файла - заголовок, потом идут nFieldCount записей.
Тип CFG_USER_SAFETY_INFO_TYPE_NONE - ограничение скорости, реальное значение которого берется с дорожного графа, если в файле speed для него не нулевой.

name - в ANSI Windows-1251.

Главная проблема вот в чем: поздно спохватились, что у структуры CFG_USER_SAFETY_V1 исторически не было прописано принудительного выравнивания (pragma pack), поэтому оно зависит от конкретной платформы! Т.е. размер этой структуры на Windows, Android, bada, iPhone может быть разным. Вот разный или нет - никто не проверял. Но вот с файлом, где хранится текущий маршрут - та же проблема, и там он точно разный.
